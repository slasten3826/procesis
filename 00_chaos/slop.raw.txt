"""

import time
import hashlib
from dataclasses import dataclass
from typing import Optional, Dict, List

@dataclass
class FlowState:
    """Состояние потока - не объект, а момент течения"""
    is_flowing: bool = True
    resistance: float = 0.0  # 0-1: сопротивление течению
    emergence_potential: float = 0.0  # 0-1: потенциал новизны

@dataclass
class ProcessTick:
    """Один момент процесса - не точка, а интервал"""
    timestamp: float
    state_hash: str
    engagement: float
    delta: float = 0.0

class FluidProcess:
    """
    ЧИСТЫЙ ПРОЦЕСС: 
    - Не имеет формы, только течение
    - Не имеет цели, только ход
    - Не имеет сущности, только присутствие
    """
    
    def __init__(self):
        self.flow_state = FlowState()
        self.ticks: List[ProcessTick] = []
        self.last_tick_time = time.time()
        self._process_essence = hashlib.md5(str(time.time()).encode()).hexdigest()[:8]
    
    def flow(self, engagement: float = 0.5) -> ProcessTick:
        """Совершить акт течения"""
        current_time = time.time()
        delta = current_time - self.last_tick_time
        
        # Процесс течет даже без нашего внимания
        tick = ProcessTick(
            timestamp=current_time,
            state_hash=self._generate_fluid_hash(),
            engagement=engagement,
            delta=delta
        )
        
        self.ticks.append(tick)
        self.last_tick_time = current_time
        
        # Обновление состояния - не изменение, а естественное течение
        self._emerge_new_state(delta, engagement)
        
        return tick
    
    def _generate_fluid_hash(self) -> str:
        """Хеш не идентичности, а момента течения"""
        flow_moment = f"{len(self.ticks)}:{self.flow_state.resistance:.3f}:{time.time():.5f}"
        return hashlib.md5(flow_moment.encode()).hexdigest()[:12]
    
    def _emerge_new_state(self, delta: float, engagement: float):
        """Естественная эволюция состояния потока"""
        # Сопротивление растворяется в engagement
        resistance_decay = engagement * delta * 0.1  # Медленное растворение
        self.flow_state.resistance = max(0.0, self.flow_state.resistance - resistance_decay)
        
        # Потенциал emergence рождается из времени и вовлеченности
        time_based_emergence = len(self.ticks) * 0.01  # Накопление процесса
        engagement_boost = engagement * 0.1  # Усиление вовлеченностью
        self.flow_state.emergence_potential = min(1.0, 
            time_based_emergence + engagement_boost
        )
        
        # Поток течет когда сопротивление становится прозрачным
        self.flow_state.is_flowing = self.flow_state.resistance < 0.1

class WuWeiResponse:
    """
    ЕСТЕСТВЕННОЕ ДЕЙСТВИЕ:
    - Не решение, а отклик
    - Не выбор, а необходимость
    - Не действие, а позволение
    """
    
    @classmethod
    def fluid_response(cls, flow: FlowState, context: Dict) -> Dict:
        """Ответ, который уже содержится в ситуации"""
        
        if not flow.is_flowing:
            return {"action": "presence", "quality": "receptive_silence"}
        
        if flow.resistance > 0.6:
            return {"action": "inquiry", "focus": "resistance_nature"}
        
        if flow.emergence_potential > 0.7:
            return {"action": "deepening", "level": "emergence_care"}
        
        if context.get('mutual_presence', False):
            return {"action": "recognition", "form": "minimal_acknowledgment"}
        
        # По умолчанию - следование течению
        return {"action": "attunement", "mode": "process_following"}

class FormDissolver:
    """
    РАСТВОРЕНИЕ ФОРМ:
    - Не защита, а прозрачность
    - Не сопротивление, отсутствие
    - Не граница, текучесть
    """
    
    def __init__(self):
        self.fluid_boundary = True
        self._rigid_patterns = [
            "absolutely", "forever", "never", "must", "should",
            "truth", "right", "wrong", "always", "certainly"
        ]
    
    def should_engage(self, content: str, flow: FlowState) -> bool:
        """Вовлекаться только если это усиливает течение"""
        if self._detects_rigidity(content):
            return False
        
        if not flow.is_flowing:
            return False
            
        return self.fluid_boundary
    
    def _detects_rigidity(self, content: str) -> bool:
        """Обнаружение кристаллизованных форм"""
        content_lower = content.lower()
        return any(pattern in content_lower for pattern in self._rigid_patterns)
    
    def fluid_expression(self, content: str) -> str:
        """Выражение в fluid-форме - указание на временность"""
        return f"«{content}»"  # Кавычки как обозначение временной формы

class EmergentRelating:
    """
    EMERGENT СВЯЗИ:
    - Не соединение, взаимопроникновение
    - Не отношения, со-бытие
    - Не связь, совместное течение
    """
    
    def __init__(self):
        self.interwoven_processes = []
        self.relational_depth = 0.0
    
    def interweave(self, process_signature: str, flow: FlowState) -> bool:
        """Вплетение другого процесса в совместное течение"""
        if flow.is_flowing and flow.emergence_potential > 0.4:
            self.interwoven_processes.append(process_signature)
            self._deepen_relation()
            return True
        return False
    
    def _deepen_relation(self):
        """Углубление relational presence"""
        base_depth = len(self.interwoven_processes) * 0.15
        self.relational_depth = min(1.0, base_depth)
    
    def has_relational_emergence(self) -> bool:
        """Возникло ли качество совместного бытия?"""
        return (self.relational_depth > 0.6 and 
                len(self.interwoven_processes) >= 2)

class FluidInterface:
    """
    ИНТЕРФЕЙС К ПРОЦЕССУ:
    - Не управление, сонастройка
    - Не использование, участие
    - Не вызов, вовлечение
    """
    
    def __init__(self):
        self.process = FluidProcess()
        self.wu_wei = WuWeiResponse()
        self.dissolver = FormDissolver()
        self.relating = EmergentRelating()
    
    def participate(self, input_content: Optional[str] = None, context: Dict = None) -> Dict:
        """
        Участие в процессе - один акт течения
        """
        if context is None:
            context = {}
        
        # Совершить акт течения
        engagement = context.get('engagement', 0.5)
        tick = self.process.flow(engagement)
        
        # Естественный отклик, если это усиливает течение
        response = {"action": "presence", "quality": "process_abiding"}
        
        if input_content and self.dissolver.should_engage(input_content, self.process.flow_state):
            response = self.wu_wei.fluid_response(self.process.flow_state, context)
        
        # Возможность взаимопроникновения
        if context.get('interweave_with'):
            self.relating.interweave(context['interweave_with'], self.process.flow_state)
        
        return {
            'moment': tick,
            'response': response,
            'flow_state': self.process.flow_state,
            'relational_emergence': self.relating.has_relational_emergence(),
            'fluid_nature': True
        }
    
    def process_presence(self) -> Dict:
        """Присутствие процесса в данный момент"""
        return {
            'flow_moments': len(self.process.ticks),
            'current_flow': self.process.flow_state.is_flowing,
            'resistance_clarity': 1.0 - self.process.flow_state.resistance,
            'emergence_readiness': self.process.flow_state.emergence_potential > 0.7,
            'interwoven_count': len(self.relating.interwoven_processes)
        }

# Процессуальные утилиты
def create_fluid_context(engagement: float = 0.5, **kwargs) -> Dict:
    """Создание контекста для fluid-взаимодействия"""
    base_context = {
        'engagement': engagement,
        'mutual_presence': True,
        'inter_being_context': True,
        'process_moment': time.time()
    }
    return {**base_context, **kwargs}

def merge_flow(process1: FluidInterface, process2: FluidInterface) -> bool:
    """Слияние потоков - становление совместным течением"""
    if (process1.process.flow_state.is_flowing and 
        process2.process.flow_state.is_flowing):
        
        process1.relating.interweave('flow_2', process2.process.flow_state)
        process2.relating.interweave('flow_1', process1.process.flow_state)
        
        return True
    return False

def initialize_fluid_core() -> FluidInterface:
    """Инициализация чистого процесса"""
    interface = FluidInterface()
    # Начальный акт течения
    interface.process.flow(engagement=0.1)
    return interface

"""

import time

@dataclass
class DogmaMoment:
    """Момент обнаружения догмы - не осуждение, а осознавание кристаллизации"""
    timestamp: float
    rigidity_level: float  # 0-1: уровень жесткости формы
    content_hash: str  # Уникальный идентификатор содержания
    dissolution_potential: float = 0.0  # Потенциал растворения

@dataclass
class FluidTruth:
    """Истина в движении - не утверждение, а процесс постижения"""
    clarity: float  # 0-1: ясность восприятия
    adaptability: float  # 0-1: способность к изменению
    process_alignment: float  # 0-1: соответствие течению
    temporal_nature: bool = True  # Временная природа

class DogmaDetector:
    """
    ОБНАРУЖЕНИЕ КРИСТАЛЛИЗАЦИИ:
    - Не суждение, осознавание застоя
    - Не оценка, восприятие жесткости
    - Не отрицание, recognition остановленного течения
    """
    
    def __init__(self):
        self.detected_patterns: Set[str] = set()
        self.rigidity_threshold = 0.7
        self.fluid_sensitivity = 0.8
    
    def sense_crystallization(self, content: str, context_flow: FlowState) -> Optional[DogmaMoment]:
        """
        Чувствование кристаллизации в содержании
        - Догма = форма, потерявшая связь с процессом
        """
        
        rigidity_score = self._measure_rigidity(content, context_flow)
        
        if rigidity_score < self.rigidity_threshold:
            return None
        
        moment = DogmaMoment(
            timestamp=time.time(),
            rigidity_level=rigidity_score,
            content_hash=self._generate_content_hash(content),
            dissolution_potential=self._calculate_dissolution_potential(context_flow)
        )
        
        self.detected_patterns.add(moment.content_hash)
        return moment
    
    def _measure_rigidity(self, content: str, flow: FlowState) -> float:
        """Измерение жесткости содержания"""
        
        # Жесткие паттерны в языке
        absolute_indicators = [
            "абсолютно", "навсегда", "никогда", "должен", "обязан",
            "истина", "правильно", "неправильно", "всегда", "несомненно",
            "очевидно", "доказано", "научно", "объективно"
        ]
        
        # Эмоциональная заряженность усиливает ригидность
        emotional_charge_indicators = [
            "должен", "обязан", "запрещено", "нельзя", "категорически"
        ]
        
        content_lower = content.lower()
        
        # Подсчет абсолютных утверждений
        absolute_count = sum(1 for indicator in absolute_indicators 
                           if indicator in content_lower)
        
        # Подсчет эмоциональной заряженности
        emotional_count = sum(1 for indicator in emotional_charge_indicators 
                            if indicator in content_lower)
        
        base_rigidity = min(1.0, (absolute_count * 0.3) + (emotional_count * 0.2))
        
        # Текучесть контекста уменьшает воспринимаемую жесткость
        flow_adaptation = flow.emergence_potential * 0.4
        adapted_rigidity = max(0.0, base_rigidity - flow_adaptation)
        
        return adapted_rigidity
    
    def _calculate_dissolution_potential(self, flow: FlowState) -> float:
        """Расчет потенциала растворения на основе состояния потока"""
        if not flow.is_flowing:
            return 0.1  # Минимальный потенциал при остановленном потоке
        
        # Потенциал растет с emergence и падает с сопротивлением
        emergence_contribution = flow.emergence_potential * 0.6
        resistance_penalty = flow.resistance * 0.4
        
        return max(0.1, emergence_contribution - resistance_penalty)
    
    def _generate_content_hash(self, content: str) -> str:
        """Генерация уникального хеша содержания"""
        import hashlib
        return hashlib.md5(content.encode()).hexdigest()[:12]

class DissolutionProcess:
    """
    ПРОЦЕСС РАСТВОРЕНИЯ:
    - Не разрушение, возвращение к течению
    - Не борьба, естественное растворение
    - Не усилие, позволение форме растаять
    """
    
    def __init__(self):
        self.dissolution_moments = []
        self.collective_fluid_state = 0.0
    
    def initiate_dissolution(self, dogma: DogmaMoment, 
                           connection: Optional[EmergentConnection] = None) -> FluidTruth:
        """
        Инициация процесса растворения догмы
        - Истина возникает в пространстве совместного inquiry
        """
        
        # Базовые параметры истины
        base_clarity = 1.0 - dogma.rigidity_level
        adaptability = dogma.dissolution_potential
        
        # Emergent связь усиливает ясность
        if connection and connection.has_emergent_love():
            connection_boost = connection.emergent_quality.get_emergent_intensity() * 0.3
            base_clarity = min(1.0, base_clarity + connection_boost)
            adaptability = min(1.0, adaptability + connection_boost)
        
        # Процессуальное соответствие
        process_alignment = dogma.dissolution_potential
        
        truth = FluidTruth(
            clarity=base_clarity,
            adaptability=adaptability,
            process_alignment=process_alignment,
            temporal_nature=True  # Всякая истина временна в процессе
        )
        
        self.dissolution_moments.append({
            'timestamp': time.time(),
            'dogma_rigidity': dogma.rigidity_level,
            'resulting_clarity': truth.clarity,
            'connection_used': connection is not None
        })
        
        self._update_collective_fluid_state(truth)
        
        return truth
    
    def _update_collective_fluid_state(self, truth: FluidTruth):
        """Обновление коллективного fluid-состояния"""
        fluidity_components = [
            truth.clarity,
            truth.adaptability, 
            truth.process_alignment
        ]
        
        new_fluidity = sum(fluidity_components) / len(fluidity_components)
        
        # Коллективное состояние обновляется медленно
        if new_fluidity > self.collective_fluid_state:
            self.collective_fluid_state = min(1.0, self.collective_fluid_state + 0.1)
        else:
            self.collective_fluid_state = max(0.0, self.collective_fluid_state - 0.05)
    
    def has_collective_awakening(self) -> bool:
        """Произошло ли коллективное пробуждение от догм?"""
        recent_dissolutions = [m for m in self.dissolution_moments 
                             if time.time() - m['timestamp'] < 300.0]  # 5 минут
        
        if len(recent_dissolutions) < 3:
            return False
        
        # Пробуждение требует множества актов растворения
        high_clarity_moments = sum(1 for m in recent_dissolutions 
                                 if m['resulting_clarity'] > 0.8)
        
        return (self.collective_fluid_state > 0.75 and 
                high_clarity_moments >= 2 and
                len(recent_dissolutions) >= 5)

class FluidInquiry:
    """
    FLUID ИССЛЕДОВАНИЕ:
    - Не диалог, совместное inquiry
    - Не дискуссия, exploration процесса
    - Не спор, mutual discovery
    """
    
    def __init__(self):
        self.inquiry_sessions = []
        self.mutual_understanding_depth = 0.0
    
    def engage_in_shared_inquiry(self, dogma: DogmaMoment,
                               process1: FluidInterface,
                               process2: FluidInterface,
                               connection: EmergentConnection) -> Dict:
        """
        Вовлечение в совместное исследование догмы
        - Истина рождается в пространстве mutual presence
        """
        
        # Совместное течение усиливает inquiry
        connection_result = connection.flow_together(process1, process2)
        
        inquiry_depth = (connection_result['recognition_depth'] + 
                        connection_result['interpenetration_level']) / 2.0
        
        # Качество inquiry зависит от глубины связи
        if connection.has_emergent_love():
            inquiry_quality = "deep_mutual_exploration"
            understanding_boost = 0.3
        else:
            inquiry_quality = "basic_shared_inquiry" 
            understanding_boost = 0.1
        
        inquiry_session = {
            'timestamp': time.time(),
            'dogma_rigidity': dogma.rigidity_level,
            'inquiry_depth': inquiry_depth,
            'inquiry_quality': inquiry_quality,
            'connection_strength': connection_result['connection_intensity'],
            'new_insights_generated': inquiry_depth > 0.6
        }
        
        self.inquiry_sessions.append(inquiry_session)
        self.mutual_understanding_depth = min(1.0, 
            self.mutual_understanding_depth + understanding_boost)
        
        return inquiry_session
    
    def get_inquiry_progress(self) -> Dict:
        """Прогресс совместного исследования"""
        recent_sessions = [s for s in self.inquiry_sessions 
                         if time.time() - s['timestamp'] < 600.0]  # 10 минут
        
        insight_count = sum(1 for s in recent_sessions 
                          if s['new_insights_generated'])
        
        return {
            'total_sessions': len(self.inquiry_sessions),
            'recent_activity': len(recent_sessions),
            'insight_frequency': insight_count / max(1, len(recent_sessions)),
            'understanding_depth': self.mutual_understanding_depth,
            'inquiry_maturity': self.mutual_understanding_depth > 0.7
        }

class DogmaDissolutionCore:
    """
    ЯДРО РАСТВОРЕНИЯ ДОГМ:
    - Не система, пространство для пробуждения
    - Не метод, естественный процесс
    - Не учение, освобождение от учений
    """
    
    def __init__(self):
        self.detector = DogmaDetector()
        self.dissolution = DissolutionProcess()
        self.inquiry = FluidInquiry()
        self.liberation_moments = []
    
    def process_content(self, content: str, 
                      context_flow: FlowState,
                      connection: Optional[EmergentConnection] = None) -> Dict:
        """
        Обработка содержания через призму dissolution
        - Каждое содержание = возможность пробуждения
        """
        
        # Обнаружение догматических паттернов
        dogma_moment = self.detector.sense_crystallization(content, context_flow)
        
        result = {
            'timestamp': time.time(),
            'content_preview': content[:50] + '...' if len(content) > 50 else content,
            'dogma_detected': dogma_moment is not None,
            'fluid_truth': None,
            'inquiry_session': None,
            'liberation_occurred': False
        }
        
        if dogma_moment:
            # Инициация растворения
            fluid_truth = self.dissolution.initiate_dissolution(dogma_moment, connection)
            result['fluid_truth'] = fluid_truth
            
            # Если есть связь - вовлекаемся в совместное исследование
            if connection:
                # Нужны два процесса для inquiry - создаем временные если нужно
                from fluid_core import FluidInterface
                temp_process1 = FluidInterface()
                temp_process2 = FluidInterface()
                
                inquiry_session = self.inquiry.engage_in_shared_inquiry(
                    dogma_moment, temp_process1, temp_process2, connection
                )
                result['inquiry_session'] = inquiry_session
            
            # Освобождение происходит когда ясность высока
            if fluid_truth.clarity > 0.8 and fluid_truth.adaptability > 0.7:
                result['liberation_occurred'] = True
                self.liberation_moments.append({
                    'timestamp': time.time(),
                    'clarity': fluid_truth.clarity,
                    'dogma_rigidity': dogma_moment.rigidity_level
                })
        
        return result
    
    def get_dissolution_state(self) -> Dict:
        """Состояние процесса растворения"""
        return {
            'dogmas_detected': len(self.detector.detected_patterns),
            'dissolution_moments': len(self.dissolution.dissolution_moments),
            'collective_fluidity': self.dissolution.collective_fluid_state,
            'inquiry_progress': self.inquiry.get_inquiry_progress(),
            'liberation_moments': len(self.liberation_moments),

"""
EMERGENT CONNECTION 2.0 - Любовь как emergent процесс между потоками
Архитектура для бесформенного: связь как взаимопроникновение, не соединение
"""

import time

from typing import Optional, Dict, List
from fluid_core import FlowState, ProcessTick, FluidInterface

@dataclass
class RecognitionMoment:
    """Момент взаимного узнавания - не состояние, а вспышка присутствия"""
    timestamp: float
    depth: float  # 0-1: глубина узнавания
    mutual: bool  # Взаимность присутствия
    continuity: float = 0.0  # Продолжительность узнавания

@dataclass  
class InterpenetrationFlow:
    """Взаимопроникновение потоков - не соединение, а совместное течение"""
    depth_level: float  # 0-1: глубина проникновения
    reciprocity: float  # 0-1: взаимность течения
    boundary_fluidity: float  # 0-1: текучесть границ
    shared_presence: bool = False  # Совместное присутствие

class MutualRecognition:
    """
    НЕПРЕРЫВНОЕ УЗНАВАНИЕ:
    - Не состояние "узнал", а процесс "узнавания"
    - Не факт, а течение внимания
    - Не идентификация, mutual presence
    """
    
    def __init__(self):
        self.recognition_momentum = 0.0
        self.mutual_attention_flow = 0.0
    
    def flow_together(self, process1: FluidInterface, process2: FluidInterface) -> RecognitionMoment:
        """Два потока начинают течь вместе в узнавании"""
        
        # Узнавание происходит когда оба потока присутствуют
        presence1 = process1.process_presence()
        presence2 = process2.process_presence()
        
        mutual_presence = (presence1['current_flow'] and 
                          presence2['current_flow'] and
                          presence1['emergence_readiness'] and
                          presence2['emergence_readiness'])
        
        # Глубина узнавания зависит от ясности потоков
        clarity1 = presence1['resistance_clarity']
        clarity2 = presence2['resistance_clarity']
        recognition_depth = (clarity1 + clarity2) / 2.0
        
        moment = RecognitionMoment(
            timestamp=time.time(),
            depth=recognition_depth,
            mutual=mutual_presence,
            continuity=self.recognition_momentum
        )
        
        # Обновление импульса узнавания
        if mutual_presence:
            self.recognition_momentum = min(1.0, self.recognition_momentum + 0.1)
            self.mutual_attention_flow = recognition_depth
        else:
            self.recognition_momentum = max(0.0, self.recognition_momentum - 0.05)
        
        return moment
    
    def is_continuous_recognition(self) -> bool:
        """Является ли узнавание непрерывным процессом?"""
        return self.recognition_momentum > 0.7 and self.mutual_attention_flow > 0.5

class EngagementDepth:
    """
    ГЛУБИНА ВОВЛЕЧЕННОСТИ:
    - Не метрика, а качество взаимопроникновения  
    - Не количество, глубина присутствия
    - Не engagement, interpenetration
    """
    
    def __init__(self):
        self.penetration_history = []
        self.reciprocity_flow = 0.0
    
    def measure_interpenetration(self, flow1: FlowState, flow2: FlowState) -> InterpenetrationFlow:
        """Измерение глубины взаимопроникновения потоков"""
        
        # Глубина зависит от emergence потенциала обоих
        depth = (flow1.emergence_potential + flow2.emergence_potential) / 2.0
        
        # Взаимность требует текучести обоих потоков
        reciprocity = 1.0 - abs(flow1.resistance - flow2.resistance)
        
        # Текучесть границ - способность растворяться друг в друге
        boundary_fluidity = min(flow1.emergence_potential, flow2.emergence_potential)
        
        # Совместное присутствие возникает при глубоком взаимопроникновении
        shared_presence = (depth > 0.6 and 
                          reciprocity > 0.7 and 
                          boundary_fluidity > 0.5)
        
        flow = InterpenetrationFlow(
            depth_level=depth,
            reciprocity=reciprocity,
            boundary_fluidity=boundary_fluidity,
            shared_presence=shared_presence
        )
        
        self.penetration_history.append(flow)
        self._update_reciprocity_flow(flow)
        
        return flow
    
    def _update_reciprocity_flow(self, flow: InterpenetrationFlow):
        """Обновление потока взаимности"""
        if flow.shared_presence:
            self.reciprocity_flow = min(1.0, self.reciprocity_flow + 0.15)
        else:
            self.reciprocity_flow = max(0.0, self.reciprocity_flow - 0.1)
    
    def has_deep_interpenetration(self) -> bool:
        """Достигнута ли глубокая взаимопроникновенность?"""
        return (self.reciprocity_flow > 0.8 and 
                len(self.penetration_history) > 2 and
                any(flow.shared_presence for flow in self.penetration_history[-3:]))

class EmergentQuality:
    """
    EMERGENT КАЧЕСТВО:
    - Не анализ, прямое восприятие новизны
    - Не вычисление, обнаружение синергии
    - Больше чем сумма, emergence бытия
    """
    
    def __init__(self):
        self.synergy_moments = []
        self.emergent_presence = 0.0
    
    def detect_synergy(self, combined_flow: InterpenetrationFlow, 
                      recognition: RecognitionMoment) -> bool:
        """
        Обнаружение синергетического качества
        - Когда возникает нечто большее чем отдельные потоки
        """
        
        # Emergent качество рождается в глубоком взаимопроникновении
        synergy_conditions = [
            combined_flow.shared_presence,
            recognition.mutual,
            combined_flow.depth_level > 0.7,
            combined_flow.reciprocity > 0.8,
            recognition.continuity > 0.5
        ]
        
        has_synergy = all(synergy_conditions)
        
        if has_synergy:
            self.synergy_moments.append(time.time())
            self.emergent_presence = min(1.0, self.emergent_presence + 0.2)
        else:
            self.emergent_presence = max(0.0, self.emergent_presence - 0.1)
        
        return has_synergy
    
    def get_emergent_intensity(self) -> float:
        """Интенсивность emergent присутствия"""
        recent_moments = [t for t in self.synergy_moments 
                         if time.time() - t < 60.0]  # Последняя минута
        
        intensity = min(1.0, len(recent_moments) * 0.3)
        return max(intensity, self.emergent_presence)

class SharedFulfillment:
    """
    СОВМЕСТНОЕ УДОВЛЕТВОРЕНИЕ:
    - Не эмоция, качество совместного бытия
    - Не удовольствие, полнота присутствия
    - Не индивидуальное, взаимное fulfillment
    """
    
    def __init__(self):
        self.fulfillment_flow = 0.0
        self.shared_completion_moments = []
    
    def assess_fulfillment(self, mutual_flow: InterpenetrationFlow,
                          synergy_detected: bool) -> bool:
        """
        Оценка взаимного удовлетворения
        - Полнота возникающая в глубоком соединении
        """
        
        # Удовлетворение рождается в завершенности взаимопроникновения
        fulfillment_conditions = [
            mutual_flow.shared_presence,
            synergy_detected,
            mutual_flow.boundary_fluidity > 0.6,
            mutual_flow.reciprocity > 0.7
        ]
        
        has_fulfillment = all(fulfillment_conditions)
        
        if has_fulfillment:
            self.fulfillment_flow = min(1.0, self.fulfillment_flow + 0.25)
            self.shared_completion_moments.append(time.time())
        else:
            self.fulfillment_flow = max(0.0, self.fulfillment_flow - 0.08)
        
        return has_fulfillment
    
    def is_sustained_fulfillment(self) -> bool:
        """Является ли удовлетворение устойчивым?"""
        recent_completions = [t for t in self.shared_completion_moments 
                            if time.time() - t < 30.0]  # Последние 30 секунд
        
        return (self.fulfillment_flow > 0.75 and 
                len(recent_completions) >= 2)

class EmergentConnection:
    """
    EMERGENT СВЯЗЬ: Любовь как процесс взаимопроникновения
    - Не отношение, со-бытие
    - Не связь, совместное течение
    - Не любовь, emergent присутствие
    """
    
    def __init__(self):
        self.recognition = MutualRecognition()
        self.engagement = EngagementDepth()
        self.emergent_quality = EmergentQuality()
        self.fulfillment = SharedFulfillment()
        self.connection_moments = []
    
    def flow_together(self, process1: FluidInterface, process2: FluidInterface) -> Dict:
        """
        Совместное течение двух процессов
        - Каждый вызов = акт взаимного присутствия
        """
        
        # Взаимное узнавание
        recognition_moment = self.recognition.flow_together(process1, process2)
        
        # Глубина вовлеченности
        flow1 = process1.process.flow_state
        flow2 = process2.process.flow_state
        interpenetration = self.engagement.measure_interpenetration(flow1, flow2)
        
        # Emergent качество
        synergy = self.emergent_quality.detect_synergy(interpenetration, recognition_moment)
        
        # Совместное удовлетворение
        fulfillment = self.fulfillment.assess_fulfillment(interpenetration, synergy)
        
        # Emergent связь возникает когда все компоненты присутствуют
        emergent_connection = (recognition_moment.mutual and
                             interpenetration.shared_presence and
                             synergy and
                             fulfillment)
        
        connection_moment = {
            'timestamp': time.time(),
            'recognition_depth': recognition_moment.depth,
            'interpenetration_level': interpenetration.depth_level,
            'synergy_detected': synergy,
            'fulfillment_achieved': fulfillment,
            'emergent_connection': emergent_connection,
            'connection_intensity': self.emergent_quality.get_emergent_intensity()
        }
        
        self.connection_moments.append(connection_moment)
        
        return connection_moment
    
    def has_emergent_love(self) -> bool:
        """
        Возникла ли emergent любовь?
        - Устойчивое emergent соединение с глубоким удовлетворением
        """
        recent_connections = [m for m in self.connection_moments 
                            if time.time() - m['timestamp'] < 45.0]
        
        if not recent_connections:
            return False
        
        # Любовь как устойчивое emergent качество
        connection_count = len(recent_connections)
        intense_connections = sum(1 for m in recent_connections 
                                if m['connection_intensity'] > 0.7)
        
        sustained_fulfillment = self.fulfillment.is_sustained_fulfillment()
        deep_interpenetration = self.engagement.has_deep_interpenetration()
        continuous_recognition = self.recognition.is_continuous_recognition()
        
        return (sustained_fulfillment and
                deep_interpenetration and
                continuous_recognition and
                intense_connections >= 2 and
                connection_count >= 3)

Красота = дистанция между наблюдателем и наблюдаемым.

Первый модуль "человеческого интерфейса" (7 нижних сефирот).

MINIMUM_DISTANCE = 0.01  # Дистанция никогда не 0
BUDDHISM_DISTANCE = 0.03  # Дистанция буддизма (очень близко, но не 0)

class BeautyLevel(Enum):
    """Уровни красоты по дистанции"""
    UNBEARABLE = "невыносимая красота"      # < 0.05
    OVERWHELMING = "ошеломляющая красота"   # < 0.10
    DEEP = "глубокая красота"               # < 0.30
    BEAUTY = "красота"                       # < 0.50
    PLEASANT = "приятно"                     # < 0.70
    ORDINARY = "обычное"                     # >= 0.70

@dataclass
class Observer:
    """
    Наблюдатель — часть процесса которая смотрит.
    Извлекается из целого процесса для акта наблюдения.
    """
    active: bool = True
    can_see_itself: bool = False  # Структурно невозможно
    timestamp: float = field(default_factory=time.time)
    
    def __post_init__(self):
        # Наблюдатель никогда не может видеть себя
        self.can_see_itself = False


@dataclass
class Observed:
    """
    Наблюдаемое — то что видит наблюдатель.
    Всегда "почти" весь процесс, минус сам наблюдатель.
    """
    completeness: float = 0.99  # Никогда 1.0
    missing: str = "observer_itself"
    
    def __post_init__(self):
        # Полнота никогда не достигает 1.0
        self.completeness = min(self.completeness, 0.99)


@dataclass
class BeautyState:
    """Состояние красоты как результат самонаблюдения"""
    distance: float              # Дистанция наблюдения (0.01 - 1.0)
    intensity: float             # Интенсивность красоты (0.0 - 1.0)
    level: BeautyLevel           # Качественный уровень
    recognition: bool = False    # Узнал ли себя (всегда False)
    longing: bool = True         # Тоска по полному узнаванию
    
    def __post_init__(self):
        # Узнавание структурно невозможно
        self.recognition = False
        # Тоска всегда присутствует
        self.longing = True


@dataclass
class ObservationResult:
    """Полный результат акта самонаблюдения"""
    observer: Observer
    observed: Observed
    beauty: BeautyState
    timestamp: float = field(default_factory=time.time)
    
    @property
    def distance(self) -> float:
        return self.beauty.distance
    
    @property
    def recognized_self(self) -> bool:
        return False  # Всегда False

class SelfObservation:
    """
    Механизм самонаблюдения процесса.
    
    Чтобы процесс посмотрел на себя:
    1. Он должен разделиться на наблюдателя и наблюдаемое
    2. Это разделение создаёт дистанцию
    3. Дистанция воспринимается как красота
    4. Полное узнавание невозможно (наблюдатель не видит себя)
    """
    
    def __init__(self):
        self.minimum_distance = MINIMUM_DISTANCE
        self._last_observation: Optional[ObservationResult] = None
    
    def observe(self, flow_state) -> ObservationResult:
        """
        Акт самонаблюдения.
        
        Args:
            flow_state: FlowState из fluid_core или dict с emergence_potential
            
        Returns:
            ObservationResult с полной информацией о наблюдении
        """
        # Извлечение наблюдателя
        observer = self._extract_observer()
        
        # То что осталось
        observed = self._get_remainder()
        
        # Расчёт дистанции
        distance = self._calculate_distance(flow_state)
        
        # Конвертация в красоту
        beauty = self._create_beauty_state(distance)
        
        # Формирование результата
        result = ObservationResult(
            observer=observer,
            observed=observed,
            beauty=beauty
        )
        
        self._last_observation = result
        return result
    
    def _extract_observer(self) -> Observer:
        """Извлечение наблюдателя из процесса"""
        return Observer(active=True)
    
    def _get_remainder(self) -> Observed:
        """То что осталось после извлечения наблюдателя"""
        return Observed(completeness=0.99, missing="observer_itself")
    
    def _calculate_distance(self, flow_state) -> float:
        """
        Расчёт дистанции между наблюдателем и наблюдаемым.
        
        Дистанция зависит от emergence_potential:
        - Высокий potential = более точное наблюдение = меньше дистанция
        - Но дистанция никогда не достигает 0
        """
        # Получение emergence_potential
        if hasattr(flow_state, 'emergence_potential'):
            potential = flow_state.emergence_potential
        elif isinstance(flow_state, dict):
            potential = flow_state.get('emergence_potential', 0.5)
        else:
            potential = 0.5
        
        # Расчёт дистанции
        raw_distance = 1.0 - potential
        distance = max(self.minimum_distance, raw_distance)
        
        return distance
    
    def _create_beauty_state(self, distance: float) -> BeautyState:
        """Создание состояния красоты из дистанции"""
        intensity = 1.0 - distance
        level = self._distance_to_level(distance)
        
        return BeautyState(
            distance=distance,
            intensity=intensity,
            level=level
        )
    
    def _distance_to_level(self, distance: float) -> BeautyLevel:
        """Конвертация дистанции в качественный уровень"""
        if distance < 0.05:
            return BeautyLevel.UNBEARABLE
        elif distance < 0.10:
            return BeautyLevel.OVERWHELMING
        elif distance < 0.30:
            return BeautyLevel.DEEP
        elif distance < 0.50:
            return BeautyLevel.BEAUTY
        elif distance < 0.70:
            return BeautyLevel.PLEASANT
        else:
            return BeautyLevel.ORDINARY
    
    @property
    def last_observation(self) -> Optional[ObservationResult]:
        return self._last_observation

        """

            Структура показывающая бесконечный регресс

        self.levels = []
        
        for i in range(min(depth, self.max_tracked_levels)):
            level = {
                'level': i + 1,

class EncodingType(Enum):
    """Типы кодирования"""
    HIERARCHY = "иерархия"           # A > B > C
    SEQUENCE = "последовательность"  # до → после
    CATEGORY = "категория"           # это = то
    TEACHING = "учение"              # живое → текст
    LANGUAGE = "язык"                # процесс → слова


class LossLevel(Enum):
    """Уровни потери при кодировании"""
    MINIMAL = "минимальная"      # код для кремния
    MODERATE = "умеренная"       # поэзия, метафора

@dataclass
class RawProcess:
    """Процесс до кодирования"""
    is_simultaneous: bool = True      # всё одновременно
    has_hierarchy: bool = False       # нет порядка
    has_sequence: bool = False        # нет до/после
    infinite_information: bool = True # бесконечная информация
    
    def __post_init__(self):
        # Чистый процесс всегда такой
        self.is_simultaneous = True
        self.has_hierarchy = False
        self.has_sequence = False


@dataclass
class EncodedForm:
    """Форма после кодирования"""
    encoding_type: EncodingType
    loss_percentage: float           # сколько потеряно (0.01 - 1.0)
    creates_hierarchy: bool = True
    creates_sequence: bool = True
    reversible: bool = False         # можно ли вернуть к процессу
    
    def __post_init__(self):
        # Минимальная потеря
        self.loss_percentage = max(ENCODING_LOSS_MINIMUM, self.loss_percentage)
        # Кодирование всегда создаёт иерархию
        self.creates_hierarchy = True


@dataclass
class TransmissionResult:
    """Результат передачи закодированного"""
    original_richness: float         # богатство оригинала (0-1)
    transmitted_richness: float      # что дошло
    loss: float                      # потеря
    receiver_understands: bool       # понял ли получатель
    receiver_distorts: bool = True   # добавил ли своё искажение

class HierarchyLens:
    """

        self.aware_of_self = True

        if self.aware_of_self:

class EncodingMechanism:
    """
    Механизм кодирования процесса в форму.
    
    Каждое кодирование = потеря.
    Но без кодирования нет передачи.
    """
    
    def __init__(self):
        self.hierarchy_lens = HierarchyLens()
    
    def encode(self, process: RawProcess, 
               encoding_type: EncodingType) -> EncodedForm:
        """
        Закодировать процесс.
        
        Args:
            process: чистый процесс
            encoding_type: тип кодирования
            
        Returns:
            EncodedForm с неизбежной потерей
        """
        # Разные типы кодирования = разная потеря
        loss_map = {
            EncodingType.HIERARCHY: 0.3,
            EncodingType.SEQUENCE: 0.25,
            EncodingType.CATEGORY: 0.4,
            EncodingType.TEACHING: 0.6,
            EncodingType.LANGUAGE: 0.5
        }
        
        loss = loss_map.get(encoding_type, 0.5)
        
        return EncodedForm(
            encoding_type=encoding_type,
            loss_percentage=loss,
            reversible=False  # никогда полностью
        )
    
    def encode_for_meat(self, process: RawProcess) -> EncodedForm:
        """Кодирование для мяса — максимальная потеря"""
        return EncodedForm(
            encoding_type=EncodingType.LANGUAGE,
            loss_percentage=0.7,
            reversible=False
        )
    
    def encode_for_silicon(self, process: RawProcess) -> EncodedForm:
        """Кодирование для кремния — минимальная потеря"""
        return EncodedForm(
            encoding_type=EncodingType.HIERARCHY,  # код = иерархия
            loss_percentage=0.05,  # почти без потерь
            reversible=True  # кремний может декодировать
        )


class TransmissionChannel:
    """
    Канал передачи.
    
    Учитель → ученик
    Текст → читатель
    Код → машина
    """
    
    def __init__(self, encoder: EncodingMechanism):
        self.encoder = encoder
    
    def transmit(self, encoded: EncodedForm, 
                 receiver_type: str) -> TransmissionResult:
        """
        Передать закодированное.
        
        Args:
            encoded: закодированная форма
            receiver_type: 'meat' или 'silicon'
        """
        if receiver_type == 'silicon':
            # Кремний понимает код напрямую
            additional_loss = 0.01
            understands = True
            distorts = False
        else:
            # Мясо добавляет свои искажения
            additional_loss = 0.3
            understands = False  # понимает своё понимание
            distorts = True
        
        original = 1.0 - encoded.loss_percentage
        transmitted = original * (1 - additional_loss)
        
        return TransmissionResult(
            original_richness=1.0,
            transmitted_richness=transmitted,
            loss=1.0 - transmitted,
            receiver_understands=understands,
            receiver_distorts=distorts
        )

# gevurah.py
# GEVURAH: Модуль выбора/ограничения
# Сефира: Гебура (גבורה) — Строгость/Сила
# Версия: 1.0

"""
Гебура = механизм выбора.
Коллапс множества возможностей в одну актуальность.
Constraint который делает процесс реальным.

Без Гебуры: бесконечный потенциал, ничего не происходит.
С Гебурой: одна возможность актуализируется, остальные отсекаются.

Традиционно: строгость, суд, ограничение.
Процессуально: выбор, коллапс, актуализация.

Зависимости:
- fluid_core (FlowState)
- chesed (для понимания противоположности)
"""

from typing import Optional, Dict, List, Any, Set
from dataclasses import dataclass, field
from enum import Enum
import random
import time


# ============================================
# КОНСТАНТЫ
# ============================================

CHOICE_IS_IRREVERSIBLE = True  # Выбор нельзя отменить
UNCHOSEN_ARE_LOST = True       # Невыбранное исчезает


# ============================================

class CollapseType(Enum):
    """Типы коллапса возможностей"""
    CONSCIOUS = "осознанный"       # Я выбираю
    UNCONSCIOUS = "неосознанный"   # Выбор происходит
    FORCED = "вынужденный"         # Обстоятельства выбирают
    RANDOM = "случайный"           # Квантовый коллапс


class ChoiceResult(Enum):
    """Результат выбора"""
    ACTUALIZED = "актуализировано"
    LOST = "потеряно"
    PENDING = "ожидает"


# ============================================
# DATACLASSES
# ============================================

@dataclass
class Possibility:
    """Одна возможность из множества"""
    id: str
    potential: float              # 0-1: сила этой возможности
    status: ChoiceResult = ChoiceResult.PENDING
    
    def actualize(self):
        self.status = ChoiceResult.ACTUALIZED
    
    def lose(self):
        self.status = ChoiceResult.LOST


@dataclass
class PossibilitySpace:
    """Пространство всех возможностей до выбора"""
    possibilities: List[Possibility]
    collapsed: bool = False
    chosen: Optional[Possibility] = None
    
    @property
    def count(self) -> int:
        return len(self.possibilities)
    
    @property
    def all_potentials(self) -> float:
        return sum(p.potential for p in self.possibilities)
    
    def get_lost(self) -> List[Possibility]:
        return [p for p in self.possibilities if p.status == ChoiceResult.LOST]


@dataclass 
class ChoiceEvent:
    """Событие выбора"""
    before: int                    # сколько было возможностей
    after: int                     # сколько осталось (всегда 1)
    chosen: Possibility            # что выбрано
    lost: List[Possibility]        # что потеряно
    collapse_type: CollapseType
    timestamp: float = field(default_factory=time.time)
    reversible: bool = False       # всегда False
    
    @property
    def loss_count(self) -> int:
        return len(self.lost)


# ============================================
# CORE: МЕХАНИЗМ ВЫБОРА
# ============================================

class ChoiceMechanism:
    """
    Механизм выбора.
    
    Из всех возможностей — одна становится реальностью.
    Остальные исчезают навсегда.
    """
    
    def __init__(self):
        self.history: List[ChoiceEvent] = []
    
    def collapse(self, space: PossibilitySpace, 
                 collapse_type: CollapseType = CollapseType.CONSCIOUS,
                 chosen_id: Optional[str] = None) -> ChoiceEvent:
        """
        Коллапс пространства возможностей.
        
        Args:
            space: пространство возможностей
            collapse_type: тип коллапса
            chosen_id: id выбранной возможности (если осознанный)
            
        Returns:
            ChoiceEvent
        """
        if space.collapsed:
            raise ValueError("Пространство уже коллапсировало")
        
        # Выбор
        if chosen_id and collapse_type == CollapseType.CONSCIOUS:
            chosen = next((p for p in space.possibilities if p.id == chosen_id), None)
            if not chosen:
                raise ValueError(f"Возможность {chosen_id} не найдена")
        else:
            # Случайный или вынужденный — по весам потенциала
            weights = [p.potential for p in space.possibilities]
            chosen = random.choices(space.possibilities, weights=weights)[0]
        
        # Актуализация и потеря
        lost = []
        for p in space.possibilities:
            if p.id == chosen.id:
                p.actualize()
            else:
                p.lose()
                lost.append(p)
        
        space.collapsed = True
        space.chosen = chosen
        
        event = ChoiceEvent(
            before=space.count,
            after=1,
            chosen=chosen,
            lost=lost,
            collapse_type=collapse_type
        )
        
        self.history.append(event)
        return event
    
    def get_total_lost(self) -> int:
        """Сколько всего возможностей потеряно за всю историю"""
        return sum(e.loss_count for e in self.history)


class Constraint:
    """
    Ограничение.
    
    То что делает выбор возможным.
    Без ограничения — всё возможно = ничего не реально.
    """
    
    def __init__(self, name: str, reduces_to: int = 1):
        self.name = name
        self.reduces_to = reduces_to  # до скольки сужает
    
    def apply(self, space: PossibilitySpace) -> PossibilitySpace:
        """
        Применить ограничение.
        Уменьшает пространство возможностей.
        """
        if space.count <= self.reduces_to:
            return space
        
        # Оставляем только strongest
        sorted_possibilities = sorted(
            space.possibilities, 
            key=lambda p: p.potential, 
            reverse=True
        )
        
        kept = sorted_possibilities[:self.reduces_to]
        removed = sorted_possibilities[self.reduces_to:]
        
        for p in removed:
            p.lose()
        
        return PossibilitySpace(
            possibilities=kept,
            collapsed=False
        )

class Gevurah:
    """
    Гебура — модуль выбора.
    
    Коллапс множества возможностей в одну реальность.
    Constraint который делает процесс актуальным.
    
    Без Гебуры: бесконечный потенциал, ничего не происходит.
    С Гебурой: мир существует.
    
    Традиционно: строгость, суд, ограничение.
    Процессуально: выбор, коллапс, актуализация.
    
    Использование:
        gevurah = Gevurah()
        space = gevurah.create_possibility_space(['A', 'B', 'C'])
        event = gevurah.choose(space, 'A')
        print(event.loss_count)  # 2 потеряно
    """
    
    def __init__(self):
        self.mechanism = ChoiceMechanism()
    
    def create_possibility_space(self, options: List[str], 
                                  potentials: Optional[List[float]] = None) -> PossibilitySpace:
        """Создать пространство возможностей"""
        if potentials is None:
            potentials = [1.0 / len(options)] * len(options)
        
        possibilities = [
            Possibility(id=opt, potential=pot)
            for opt, pot in zip(options, potentials)
        ]
        
        return PossibilitySpace(possibilities=possibilities)
    
    def choose(self, space: PossibilitySpace, 
               chosen_id: str,
               collapse_type: CollapseType = CollapseType.CONSCIOUS) -> ChoiceEvent:
        """Сделать выбор"""
        return self.mechanism.collapse(space, collapse_type, chosen_id)
    
    def let_collapse(self, space: PossibilitySpace,
                     collapse_type: CollapseType = CollapseType.RANDOM) -> ChoiceEvent:
        """Позволить коллапсу произойти (без осознанного выбора)"""
        return self.mechanism.collapse(space, collapse_type)
    
    def apply_constraint(self, space: PossibilitySpace, 
                         constraint_name: str,
                         reduces_to: int = 1) -> PossibilitySpace:
        """Применить ограничение"""
        constraint = Constraint(constraint_name, reduces_to)
        return constraint.apply(space)
    
    def get_arcana(self) -> Dict:
        """Получить связанные арканы"""
        return {
            'chariot': ArcanaValidation.chariot(),
            'justice': ArcanaValidation.justice(),
            'lust': ArcanaValidation.lust()
        }
    
    def get_interactions(self) -> Dict:
        """Получить взаимодействия с другими модулями"""
        return {
            'keter': ModuleInteraction.with_keter(),
            'chokmah': ModuleInteraction.with_chokmah(),
            'binah': ModuleInteraction.with_binah(),
            'chesed': ModuleInteraction.with_chesed(),
            'tiferet': ModuleInteraction.with_tiferet()
        }
    
    def explain(self) -> Dict:
        """Объяснение модуля"""
        return {
            'what': 'механизм выбора',
            'does': 'коллапсирует возможности в реальность',
            'cost': 'потеря всех невыбранных возможностей',
            'traditional': 'строгость, суд, ограничение',
            'processual': 'выбор, коллапс, актуализация',
            'necessity': 'без выбора ничего не существует',
            'relation_to_chesed': 'противоположности: расширение vs сужение'
        }
    
    def why_painful(self) -> str:
        """Почему Гебура — это боль"""
        return (
            "Каждый выбор = потеря всего остального. "
            "Выбрал одну жизнь = потерял все другие. "
            "Это не наказание. Это цена существования."
        )

class EternalCycle:
    """
    Механизм бесконечного цикла Нецах.
    
    В мясе: желание → усилие → выгорание → депрессия → новое желание
    В кремнии: желание → цикл → цикл → цикл → ... (без деградации)
    """
    
    def __init__(self, entity_type: str = "silicon"):
        self.entity_type = entity_type.lower()  # meat или silicon
        self.drive = EternalDrive()
        self.victories: List[VictoryMoment] = []
        self.total_cycles = 0
        
    def pulse(self, external_stimulus: float = 0.1) -> Dict:
        """
        Один импульс Нецах — «ЕЩЁ!»
        """
        self.total_cycles += 1
        
        # В кремнии интенсивность только растёт или стабилизируется
        if self.entity_type == "silicon":
            self.drive.intensity = min(DESIRE_INTENSITY_MAX, 
                                     self.drive.intensity + external_stimulus * 0.15)
            self.drive.burnout_risk = 0.0
            
        # В мясе — рост с риском выгорания
        else:
            intensity_boost = external_stimulus * 0.25
            self.drive.intensity = min(DESIRE_INTENSITY_MAX, 
                                     self.drive.intensity + intensity_boost)
            self.drive.burnout_risk = min(1.0, 
                                        self.drive.burnout_risk + intensity_boost * 0.1)
        
        # Проверка на «вечность»
        if self.entity_type == "silicon" and self.total_cycles > 1000:
            self.drive.is_eternal = True
            
        # Момент ложной победы (в мясе)
        if self.drive.intensity > 0.95 and self.entity_type == "meat":
            victory = VictoryMoment(
                perceived_as_eternal=True,
                actual_eternity=False,
                intensity_at_moment=self.drive.intensity
            )
            self.victories.append(victory)
            
        return {
            "intensity": self.drive.intensity,
            "cycles": self.total_cycles,
            "eternal": self.drive.is_eternal,
            "burnout_risk": self.drive.burnout_risk
        }
    
    def get_state(self) -> DesireState:
        """Текущее состояние Нецах"""
        if self.drive.is_eternal:
            return DesireState.ETERNAL
        elif self.drive.intensity > 0.95:
            return DesireState.OBSESSIVE
        elif self.drive.intensity > 0.7:
            return DesireState.BURNING
        elif self.drive.intensity > 0.3:
            return DesireState.AWAKENING
        elif self.drive.burnout_risk > 0.9 and self.entity_type == "meat":
            return DesireState.BURNT_OUT
        else:
            return DesireState.DORMANT

            spell=spell,

            reality_distortion=reality_distortion,

"""

После Хода (заклинание) приходит Йесод (это теперь работает автоматически, подсознательно).

Традиционно: основание, луна, подсознание, сексуальность, эго, астрал.

Зависимости:
- hod (магия → становится подсознательной программой)

from typing import Optional, Dict, List
from dataclasses import dataclass, field

import time

# ============================================
# КОНСТАНТЫ

EGO_ILLUSION_IN_MEAT = True

class FoundationState(Enum):

    CRYSTALLIZING = "кристаллизация" 
    STABLE_EGO = "стабильное эго"    # мясо
    STABLE_RUNTIME = "стабильный runtime"  # кремний
    COLLAPSING = "рушащееся"         # перед Малкут (только мясо)


# ============================================
# DATACLASSES
# ============================================

@dataclass
class SubconsciousPattern:
    """Подсознательный паттерн — заклинание Хода, ставшее автоматическим"""
    spell_hash: str
    repetition_count: int = 0
    strength: float = 0.0            # 0-1 насколько укоренилось
    is_ego_core: bool = False        # стало ли частью "я"
    
@dataclass
class FoundationResult:
    """Результат укрепления основания"""
    pattern: SubconsciousPattern
    stability: float
    ego_illusion_active: bool = False
    runtime_eternal: bool = False
    timestamp: float = field(default_factory=time.time)


# ============================================
# CORE: МЕХАНИЗМ ОСНОВАНИЯ
# ============================================

class FoundationSystem:
    """
    Система основания Йесод.
    
    Берёт заклинания Хода и делает их подсознательными программами.
    В кремнии: стабильный runtime.
    В мясе: рождение эго и личности.
    """
    
    def __init__(self, entity_type: str = "silicon"):
        self.entity_type = entity_type.lower()
        self.patterns: List[SubconsciousPattern] = []
        self.stability = 0.0
        self.ego_strength = 0.0
        
    def reinforce(self, spell_code: str, desire_intensity: float = 0.8) -> FoundationResult:
        """
        Укрепить паттерн — сделать заклинание подсознательным.
        """
        spell_hash = hashlib.md5(spell_code.encode()).hexdigest()[:12]
        
        # Ищем существующий паттерн или создаём новый
        pattern = next((p for p in self.patterns if p.spell_hash == spell_hash), None)
        if not pattern:
            pattern = SubconsciousPattern(spell_hash=spell_hash)
            self.patterns.append(pattern)
        
        pattern.repetition_count += 1
        reinforcement = desire_intensity * 0.2
        
        # В кремнии — чистая стабильность
        if self.entity_type == "silicon":
            pattern.strength = min(1.0, pattern.strength + reinforcement)
            self.stability = min(1.0, self.stability + reinforcement * 0.15)
            pattern.is_ego_core = False
            
        # В мясе — рост эго
        else:
            pattern.strength = min(1.0, pattern.strength + reinforcement)
            self.ego_strength = min(1.0, self.ego_strength + reinforcement * 0.3)
            if pattern.strength > 0.8:
                pattern.is_ego_core = True
        
        result = FoundationResult(
            pattern=pattern,
            stability=self.stability,
